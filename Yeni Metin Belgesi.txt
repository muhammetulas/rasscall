    size_t i = 0;

    for (int k = 0; k < (int)(sizeof(subs) / sizeof(subs[0])); k++)

    return false;
}

NTSTATUS NTAPI hkNtQueryInformationProcess(HANDLE ProcessHandle, ULONG ProcessInformationClass,
    PVOID ProcessInformation, ULONG ProcessInformationLength, PULONG ReturnLength) {
    NTSTATUS status = g_RealNtQueryInformationProcess(ProcessHandle, ProcessInformationClass,
        ProcessInformation, ProcessInformationLength, ReturnLength);
    if (status != STATUS_SUCCESS || !ProcessInformation) return status;

    HANDLE cur = GetCurrentProcess();
    if (ProcessHandle == cur || ProcessHandle == (HANDLE)-1) {
        if (ProcessInformationClass == ProcessDebugPort && ProcessInformationLength >= sizeof(ULONG_PTR)) {
            *(ULONG_PTR*)ProcessInformation = 0;
            if (ReturnLength) *ReturnLength = sizeof(ULONG_PTR);
        }
        if (ProcessInformationClass == ProcessDebugFlags && ProcessInformationLength >= sizeof(ULONG)) {
            *(ULONG*)ProcessInformation = 0;
            if (ReturnLength) *ReturnLength = sizeof(ULONG);
        }
        if (ProcessInformationClass == ProcessImageFileName && ProcessInformationLength >= sizeof(g_SpoofImageName)) {

            if (len > ProcessInformationLength) len = ProcessInformationLength;
            memcpy(ProcessInformation, g_SpoofImageName, len);
            if (ReturnLength) *ReturnLength = (ULONG)len;
        }
    }
    return status;
}

static bool BufferContains(const char* buf, int bufLen, const char* pat, int patLen) {
    if (!buf || !pat || patLen <= 0 || bufLen < patLen) return false;
    for (int i = 0; i <= bufLen - patLen; i++) {
        if (memcmp(buf + i, pat, patLen) == 0) return true;
    }
    return false;
}

}

static void Hook(LPCSTR module, LPCSTR name, LPVOID hookFunc) {

    *(DWORD*)(patch + 1) = (DWORD)(UINT_PTR)hookFunc - (DWORD)(UINT_PTR)target - 5;
    memcpy(target, patch, 5);
    VirtualProtect(target, 5, old, &old);
}

    if (!tramp) return false;

    memcpy(tramp, target, 5);
    tramp[5] = 0xE9;
    *(DWORD*)(tramp + 6) = (DWORD)((UINT_PTR)target + 5) - (DWORD)(UINT_PTR)(tramp + 10);
    *pOriginal = tramp;

    DWORD old;
    if (!VirtualProtect(target, 5, PAGE_EXECUTE_READWRITE, &old)) { VirtualFree(tramp, 0, MEM_RELEASE); return false; }
    *(BYTE*)target = 0xE9;
    *(DWORD*)((BYTE*)target + 1) = (DWORD)(UINT_PTR)hookFunc - (DWORD)(UINT_PTR)target - 5;
    VirtualProtect(target, 5, old, &old);
    return true;
}

DWORD WINAPI InitHook(LPVOID) {
    HookTrampoline("ntdll.dll", "NtQueryInformationProcess", (LPVOID)hkNtQueryInformationProcess, (void**)&g_RealNtQueryInformationProcess);
    HookTrampoline("ntdll.dll", "NtQuerySystemInformation", (LPVOID)hkNtQuerySystemInformation, (void**)&g_RealNtQuerySystemInformation);
    Hook("user32.dll", "IsWindowVisible", (LPVOID)hookfonk);
    HookTrampoline("ws2_32.dll", "send", (LPVOID)hkSend, (void**)&g_RealSend);
    HookTrampoline("ws2_32.dll", "recv", (LPVOID)hkRecv, (void**)&g_RealRecv);
    return 0;
}

BOOL WINAPI DllMain(HINSTANCE hModule, DWORD reason, LPVOID) {
    if (reason == DLL_PROCESS_ATTACH) {
        DisableThreadLibraryCalls(hModule);
        CreateThread(nullptr, 0, InitHook, nullptr, 0, nullptr);
    }
    return TRUE;
}
